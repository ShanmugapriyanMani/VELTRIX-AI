# Conviction-scaled lot sizing
THRESHOLD = {
    'TRENDING': 1.75,
    'RANGEBOUND': 2.0,
    'VOLATILE': 2.5
}

def size_scale(score_diff, regime):
    threshold = THRESHOLD[regime]
    excess = score_diff - threshold
    # 0.0 excess -> 0.5x | 2.0+ excess -> 1.0x
    scale = min(1.0, 0.5 + (excess / 2.0) * 0.5)
    return max(0.5, scale)

def calc_lots(deploy_cap, premium, score_diff, regime):
    base = deploy_cap / (premium * 65)
    scale = size_scale(score_diff, regime)
    return max(1, round(base * scale))
